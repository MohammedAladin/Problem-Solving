#include <iostream>
using namespace std;
int main() {
	int n, k;
	cin >> n >> k;
	int* c = new int[2*n+1];
	for (int i = 0; i < 2 * n + 1;i++)
	{
		cin >> c[i];
	}
	if (n != 1) {
		for (int i = 0; i < 2 * n + 1; i ++) {
			if (i % 2 == 1 && (c[i]-1) > c[i - 1] && (c[i]-1) > c[i+1])
				c[i]--;
			else
				continue;
			k--;
			if (k == 0)
				break;
		}
	}
	else {
		for (int i = 1; i < 2 * n + 1; i += 2) {
			c[i]--;
		}
	}
	for (int i = 0; i < 2 * n + 1;i++)
	{
		cout<< c[i]<<" ";
	}	
}
