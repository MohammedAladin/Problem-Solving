	cin >> n;
	bool flag = false;
	int* arr = new int[n];
	for (int i = 0; i < n;i++) {
		cin >> *(arr + i);
	}
	int s=0 , e=0;
	if (arr[0] > arr[1]) {
		for (int i = 0; i < n; i++) {
			if (arr[i] < arr[i + 1]) {
				e = i;
				flag = true;
				break;
			}
			if (!flag)
				e = n - 1;
		}
	}
	else if (arr[0] < arr[1]) {
		for (int i = 0; i < n-1; i++) {
			if (arr[i] > arr[i + 1])
			{
				s = i;
				for (int j = i; j < n - 1; j++) {
					if (arr[j] < arr[j + 1]) {
						e = j;
						flag = true;
						break;
					}
				}
				if (!flag)
					e = n - 1;
				break;
			}
		}
	}
	int start = s;
	int end = e;
	while (s < e) {
		swap(arr[s], arr[e]);
		s++;
		e--;
	}
	for (int i = 0; i < n-1;i++) {
		if (arr[i] > arr[i + 1]) {
			cout << "no";
			return 0;
		}	
	}
	cout << "yes" << endl;
	cout << start+1 <<" " << end + 1;
